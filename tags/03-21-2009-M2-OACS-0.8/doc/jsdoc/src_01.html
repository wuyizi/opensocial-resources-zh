 85    * @param {Object?} obj Any object to be looked into.
 86    * @param {string?} opt_prefix The prefix string of each output line.
 87    * @param {string?} opt_rendered The rendered string passing between recursions.
 88    * @return {string?} A tree-style string to represent the object.
 89    * @private

 91   var debugInspect = function(obj, opt_prefix, opt_rendered) {
 92     if (!opt_rendered) opt_rendered = '\n';
 93     if (!opt_prefix) opt_prefix = '';
 94     if (obj == null) return opt_rendered || "";
 95     for (var k in obj) {
 96       opt_rendered += 
 97           (opt_prefix + k + '[' + (typeof obj[k]) + '] : ' + obj[k] + '\n');
 98       if ((typeof obj[k]) == 'object') {
 99         opt_rendered = 
100             debugInspect(obj[k], opt_prefix + '\t', opt_rendered);
101       }
102     }
103     return opt_rendered || "";
104   };

155   /**
156    * Checks if the object is an array.
157    * @param {Object} obj object to be check.
158    * @return {boolean} True if the object is an array.
159    * @private
160    */
161   function isArray(obj) {
162     if (!obj) return false;
163     if (Array) {
164       return obj instanceof Array;
165     } else {
166       if ((typeof obj == 'object') && 
167           (typeof obj.length == 'number') && 
168           (typeof obj.splice != 'undefined')) {
169         return true;
170       } else {
171         return false;
172       }
173     }
174   };
175   
176   /**
177    * Iterates an array, calls the function for each item.
178    * @param {Array} array The array to be iterated.
179    * @param {Function} f The function to be applied.
180    * @param {Object} opt_thisObj The this object for the function.
181    * @private
182    */
183   function forEach(array, f, opt_thisObj) {
184     if (typeof array.length != 'number') {
185       return;
186     }
187     var l = array.length;
188     for (var i = 0; i < l; i++) {
189       if (i in array) {
190         if (false == f.call(opt_thisObj, array[i], i, array)) {
191           break;
192         };
193       }
194     }
195   };

198   /**
199    * Deeply clones an object.
200    * @param {Object?} obj The original object.
201    * @return {Object?} The cloned object.
202    * @private
203    */
204   function clone(obj) {
205     if(obj == null) return null;
206     switch (typeof obj){
207       case 'function':
208         return obj;
209       case 'object': 
210         var newObj = new obj.constructor();
211         for(var key in obj) newObj[key] = clone(obj[key]);
212         return newObj;
213       default:
214         return obj.valueOf();
215     }
216   };

218   //======================== Opensocial Data Helper Functions (private) ===========================
219   /**
220    * Gets data object from resoponse.
221    * @param {opensocial.DataResponse} dataResponse The response object.
222    * @param {string} key The data key in the response object.
223    * @throws {Error} When the RPC meets error.
224    * @return {Object?} The data, can be any instance of the opensocial.* data type.
225    * @private
226    */
227   function getData (dataResponse, key) {
228     // for debug
229     if (DEBUG) {
230       EXPORT('DATA_RESPONSE', {'key': key, 'data': dataResponse});
231     }
232     
233     if (!dataResponse) return null;
234     if (dataResponse.hadError()) {
235       var e = new Error(dataResponse.getErrorMessage());
236       e['name'] = "OpensocialError";
237       throw e;
238     }
239     if (!key) return null;
240     
241     return getDataItem(dataResponse.get(key));
242   };
243   
244   /**
245    * Gets data object from response item.
246    * @param {opensocial.ResponseItem} responseItem The response item object.
247    * @throws {Error} When the items have errors.
248    * @return {Object?} The data, can be any instance of the opensocial.* data type.
249    * @private
250    */
251   function getDataItem(responseItem) {
252     if (!responseItem) return null;
253     if (responseItem.hadError()) {
254       var e = new Error(responseItem.getErrorMessage());
255       e['name'] = "OpensocialError";
256       e['code'] = responseItem.getErrorCode();
257       throw e;
258     }
259     return responseItem.getData();
260   };

265   /**
266    * Gets the constructor function for opensocial data object.
267    * @param {Object} dataObj The object to be checked.

271   function checkType(dataObj) {

273     forEach(dataTypes, function(typeItem) {

275         type = opensocial[typeItem];
276         return false;
277       }
278     });
279     return type;

282   /**
283    * Wraps the opensocial rich object (with functions) to a JSON-like flat serializable object.
284    * So the object can pass through the <code>ExternalInterface</code> to the swf.
285    * @param {Object} dataObj The object to be wrapped.
286    * @param {Object?} opt_this The flat object as a container to hold the wrapped fields.
287    * @return {Object?} A wrapped flat object, ready to pass to swf.
288    * @private
289    */
290   function wrapObject(dataObj, opt_this) {
291     if (dataObj == null) return null;
292 
293     var type = checkType(dataObj);
294   
295     if (type == null) {
296       // treat it as string
297       return dataObj;
298     }
299    
300     // fix fields
301     opt_this = opt_this || {};
302     
303     if ('Field' in type) {    
304       opt_this['fields'] = opt_this['fields'] || {};
305 
306       // Grabs the fields data from each structures recursively.
307       for (var k in type.Field) {
308         var key = type.Field[k];
309         var value = dataObj.getField(key);
310         if (value != null) {
311           if (isArray(value)) {
312             opt_this['fields'][key] = [];
313             forEach(value, function(item) {
314               opt_this['fields'][key].push(wrapObject(item));
315             });
316           } else {
317             opt_this['fields'][key] = wrapObject(value);
318           }
319         }
320       }
321     }
322     
323     // fix properties.
324     if (type == opensocial.Person) {
325       
326       opt_this['displayName'] = dataObj.getDisplayName();
327       opt_this['isOwner'] = dataObj.isOwner();
328       opt_this['isViewer'] = dataObj.isViewer();
329       opt_this['id'] = dataObj.getId();
330       
331     } else if (type == opensocial.Activity) {
332       opt_this['id'] = dataObj.getId();
333       
334     } else if (type == opensocial.Enum) {
335       opt_this['key'] = dataObj.getKey();
336       opt_this['displayValue'] = dataObj.getDisplayValue();
337 
338     } else if (type == opensocial.Collection) {
339       opt_this['size'] = dataObj.size();
340       opt_this['offset'] = dataObj.getOffset();
341       opt_this['totalSize'] = dataObj.getTotalSize();
342       
343       var array = dataObj.asArray();
344       opt_this['array'] = [];
345       forEach(array, function(item) {
346         opt_this['array'].push(wrapObject(item));
347       });
348       
349     }
350     return opt_this || {};
351   };
352   
353   /**
354    * Unwraps the flattened object to a rich opensocial data object. The flattened object is come 
355    * from the swf via <code>ExternalInterface</code>.
356    * @param {Object} obj The object to be unwrapped. Only those 5 types with <code>setField</code>
357    *                 method and <code>opensocial.newXXXX</code> creator can be unwrapped.
358    * @param {Function} type The object's target type to be unwrapped.
359    * @throws {Error} When the type is not correct.
360    * @return {Object} Unwrapped rich opensocial data object, ready to use in javascript. 
361    * @private
362    */
363   function unwrapObject(obj, type) {
364     if ('fields' in obj) {
365       var fields = obj['fields'];
366     
367       if (type == opensocial.IdSpec) {
368         return opensocial.newIdSpec(fields);
369         
370       } else if (type == opensocial.NavigationParameters) {
371         return opensocial.newNavigationParameters(fields);
372         
373       } else if (type == opensocial.MediaItem) {
374         return opensocial.newMediaItem(fields[opensocial.MediaItem.Field.MIME_TYPE],
375                                        fields[opensocial.MediaItem.Field.URL],
376                                        fields);
377 
378       } else if (type == opensocial.Message) {
379         return opensocial.newMessage(fields[opensocial.Message.Field.BODY],
380                                      fields);
381         
382       } else if (type == opensocial.Activity) {
383         var flatItems = fields[opensocial.Activity.Field.MEDIA_ITEMS];
384         if (isArray(flatItems)) {
385           var richItems = [];
386           forEach(flatItems, function(item) {
387             richItems.push(unwrapObject(item, opensocial.MediaItem));
388           });
389           fields[opensocial.Activity.Field.MEDIA_ITEMS] = richItems;
390         }
391         return opensocial.newActivity(fields);
392       }
393     
394     }
395     // If reaches here, codes are wrong.
396     throw new Error('The object does not has fields values or type mismatched.');
397   };

475     /**
476      * Because <code>gadgets.flash.embedFlash</code> cannot auto activate it in IE6, so here we 
477      * need a special one.
478      * <p>
479      * NOTE: to be deprecated.
480      * </p>
481      * 
482      * @requires gadgets.flash
483      * 
484      * @param {string} swfUrl  The swf file's location.
485      * @param {string} swfContainer  The HTML element id.
486      * @param {string?} opt_playerVersion Required flash player version.
487      * @param {Object.<string, string>?} opt_params Flash object's parameters.
488      * @param {string?} opt_upgradeMsgHTML The message if the flashplayer version is too old.
489      * @return {boolean} True if the flash object is successfully embeded.
490      * 
491      * @member opensocial.flash
492      */
493     embedFlash: function(swfUrl, swfContainer, opt_playerVersion, opt_params, opt_upgradeMsgHTML) {
494       
495       opt_params = opt_params || {};
496       
497       // Check container element
498       var container = document.getElementById(swfContainer);
499       if (!container) return false;
500     
501       // Check version
502       var version = opt_playerVersion.match(/(\d+)(?=[\.,]?)/g);
503       var expectMajorVer = (version != null ? Number(version[0]) : 9);
504 
505       if (DEBUG) {      
506         EXPORT("currentVer", gadgets.flash.getMajorVersion());
507         EXPORT("expectVer", expectMajorVer);
508       }
509       
510       if (gadgets.flash.getMajorVersion() < expectMajorVer) {
511         opt_upgradeMsgHTML = opt_upgradeMsgHTML || 
512             '<span style="font-size:small;color:#777;"> ' +
513             'Please install or upgrade to ' +
514             '<a href="http://www.adobe.com/cn/products/flashplayer/" ' +
515             'style="color:#36C;" target="_blank">Flash Player 9.0</a>.</span>';
516         container.innerHTML = opt_upgradeMsgHTML;
517         return false;
518       };
519       
520       /*
521       opt_params['id'] = DEFAULT_FLASH_ID;
522       opt_params['name'] = DEFAULT_FLASH_ID;
523       return gadgets.flash.embedFlash(swfUrl, container, version, opt_params);
524       */
525       // Generate the flash object element
526       var html = "";
527       if (navigator.plugins && navigator.mimeTypes && navigator.mimeTypes.length) {
528         // netscape plugin architecture
529         html = '<embed src="'+ swfUrl +
530                '" id="' + DEFAULT_FLASH_ID + 
531                '" name="' + DEFAULT_FLASH_ID + 
532                '" type="application/x-shockwave-flash"';
533         for(var key in opt_params){ 
534           html += key +'="'+ opt_params[key] +'" ';
535         }
536         html += '/>';
537       } else {
538         // PC IE
539         html = '<object id="' + DEFAULT_FLASH_ID + 
540                '" classid="clsid:D27CDB6E-AE6D-11cf-96B8-444553540000"';
541         var params = '<param name="movie" value="'+ swfUrl +'" />';
542         for(var key in opt_params) {
543           if (key == 'width' || key == 'height') {
544             html += ' ' + key + '="' + opt_params[key] + '"';
545           } else {
546             params += '<param name="'+ key +'" value="'+ opt_params[key] +'" />';
547           }
548         }
549         html += '>' + params + "</object>";
550       }
551       container.innerHTML = html;
552       
553       return true;
554

877     /**
878      * Sends request to a remote site to get or post data.
879      * 
880      * @param {string} reqID The request id.
881      * @param {string} url The remote site
882      * @param {Object.<gadgets.io.RequestParameters, Object>?} opt_params 
883      *            Parameters for the request
884      * @member opensocial.flash
885      */
886     makeRequest : function(reqID, url, opt_params) {
887       opt_params = opt_params || {}
888       var postData = opt_params[gadgets.io.RequestParameters.POST_DATA];
889       if (postData) {
890         opt_params[gadgets.io.RequestParameters.POST_DATA] = gadgets.io.encodeValues(postData);
891       }
892       gadgets.io.makeRequest(url, function(obj) {
893         try {
894           if (obj['error'] || obj['rc'] != 200) {
895             var err = new Error(obj['text']);
896             err['name'] = 'OpensocialError';
897             throw err;
898           }
899           swfObj.handleMakeRequest(reqID, obj.data);
900         } catch (e) {
901           handleError(reqID, e);
902         }
903       }, opt_params);
904     },

908      * 
909      * @param {string} The service to be register.
910      * @param {string} The handler function name on the flash object.
911      * @member opensocial.flash
912      */
913     rpcRegister : function(serviceName, handlerName) {
914       if (serviceName == null) {
915         gadgets.rpc.registerDefault(swfObj[handlerName]);
916       } else {
917         gadgets.rpc.register(serviceName, function() {
918           try {
919             var args = [];
920             forEach(arguments, function(arg) {
921               args.push(arg);
922             });
923             swfObj[handlerName].apply(swfObj, args);
924           } catch(e) {
925             handleError(null, e);
926           }
927         });
928       }
929     },

945     /**
946      * Sends the rpc call request to a service on another app or container on the page.
947      * 
948      * @param {string} reqID The request id.
949      * @param {string?} targetId The target app id, null to call the parent.
950      * @param {string} serviceName The service name registered on the target.
951      * @param {Array.<Object>?} argsArray The array of args to be passed to the service.
952      * @member opensocial.flash
953      */
954     rpcCall : function(reqID, targetId, serviceName, argsArray) {
955       var reqArgs = [targetId, serviceName, function() {
956         var resArgs = [reqID];
957         forEach(arguments, function(arg) {
958           resArgs.push(arg);
959         });
960         swfObj.handleRpcCall.apply(swfObj, resArgs);
961       }];
962       
963       if (argsArray) {
964         forEach(argsArray, function(arg) {
965           reqArgs.push(arg);
966         });
967       }
968       gadgets.rpc.call.apply(null, reqArgs);
969     }
