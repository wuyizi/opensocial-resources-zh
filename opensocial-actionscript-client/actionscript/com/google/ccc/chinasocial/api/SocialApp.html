The base abstract class for all app's document class which using opensocial flash API.

NOTE:
FLEX developers don't need this base class. It is only for Adobe Flash IDE developers.


Example 
 
   // Assumed the SampleAppImpl class is the document class in your .fla file.
 
   public class SampleAppImpl extends SocialApp {
     private var data:BasicData;
     private var infoBox:TextField;
     
     public function SampleAppImpl() {
       // Do not auto-start the api.
       super(["sample"], false);
     
       // some init codes..
     
       // Create the infobox for debugging output.
       InfoBox.create(this, someX, someY);
     
       // Start the api checking and init data loading.
       api.start();
     }
     public override function setup():void {
       drawLoadingIcon();
     }
     public override function display(data:BasicData):void {
       this.data = data;
       // Draw with the loaded data.
       drawOwner(data.getOwner());
       drawViewer(data.getViewer());
     
       // Request some new data and draw.
       api.fetchFriends("OWNER_FRIENDS", null, function(ps:Collection):void {
          var arr:Array = ps.getArray();
          for (var i:int = 0;i < arr.length; i++) {
            var p:Person = arr[i];
            drawPerson(p, i);
          }
       });
     }
     public override function info(obj:Object):void {
       InfoBox.info(obj);
     }
     public override function error(e:Error):void {
       InfoBox.info(e);
     }
     
     // Other UI stuff in your app..
     private function drawPerson(....):void {
       //...
     }
     //...
   }
