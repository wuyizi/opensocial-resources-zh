 98 /**
 99  * Indicates if the opensocial library is loaded. 
100  * Initialize the <code>opensocial</code> and <code>gadgets</code> namespaces.
101  * Initialize the <code>flashAPI.swfObj</code> element. 
102  *
103  * The objectId of the swf is defined in gagdet spec. If the swf is embeded by
104  * calling <code>flashAPI.embededFlash</code> function, the detault objectId 
105  * will be 'flashAPI.swfObj'.
106  *
107  * This function is called by the swf's ExternalInterface continuingly until it
108  * returns true.
109  *
110  * @see //flash/actionscript/com/google/ccc/chinasocial/api/API.as
111  * @requires opensocial
112  * @member flashAPI
113  *
114  * @param {string?} objectId The object Id passed from the swf.
115  * @return {boolean} true if all javascripts are loaded.
116  */
117 flashAPI.isAPIReady = function(objectId) {
118   try {
119     if (!flashAPI.swfObj) {
120       flashAPI.swfObj = flashAPI.thisFlash(objectId);
121     }
122     if (flashAPI.swfObj) {
123       return true; 
124     } else {
125       return false;
126     }
127   } catch (e) {
128     INFO(e);
129     return false;
130   }
131 };

134 /**
135  * Because <code>gadgets.flash.embedFlash</code> cannot auto activate it in IE6,
136  * so here we need a special one.
137  * @requires gadgets.flash
138  * @member flashAPI
139  * @param {string} swfUrl  The swf file's location.
140  * @param {string} swfContainer  The HTML element id.
141  * @param {string?} opt_playerVersion Required flash player version.
142  * @param {Object.<string, string>?} opt_params Flash object's parameters.
143  * @param {string?} opt_upgradeMsgHTML  
144  *        The message if the browser's version needs to be upgraded.
145  */
146 flashAPI.embedFlash = function(swfUrl, swfContainer, opt_playerVersion, 
147                                opt_params, opt_upgradeMsgHTML) {
148   // Check container element
149   var container = document.getElementById(swfContainer);
150   if (!container) return false;
151 
152   // Check version
153   var version = opt_playerVersion.match(/(\d+)(?=[\.,]?)/g);
154   var expectMajorVer = (version != null ? Number(version[0]) : 9);
155   EXPORT("currentVer", gadgets.flash.getMajorVersion());
156   EXPORT("expectVer", expectMajorVer);
157   
158   if (gadgets.flash.getMajorVersion() < expectMajorVer) {
159     opt_upgradeMsgHTML = opt_upgradeMsgHTML || 
160         '<span style="font-size:small;color:#777;"> ' +
161         'Please install or upgrade to ' +
162         '<a href="http://www.adobe.com/cn/products/flashplayer/" ' +
163         'style="color:#36C;" target="_blank">Flash Player 9.0</a>.</span>';
164     container.innerHTML = opt_upgradeMsgHTML;
165     return false;
166   };
167 
168   // Generate the flash object element
169   var html = "";
170   opt_params = opt_params || {};
171   if (navigator.plugins && navigator.mimeTypes && navigator.mimeTypes.length) {
172     // netscape plugin architecture
173     html = '<embed src="'+ swfUrl +
174            '" id="' + flashAPI.DEFAULT_FLASH_ID + 
175            '" name="' + flashAPI.DEFAULT_FLASH_ID + 
176            '" type="application/x-shockwave-flash"';
177     for(var key in opt_params){ 
178       html += key +'="'+ opt_params[key] +'" ';
179     }
180     html += '/>';
181   } else {
182     // PC IE
183     html = '<object id="' + flashAPI.DEFAULT_FLASH_ID + 
184            '" classid="clsid:D27CDB6E-AE6D-11cf-96B8-444553540000"';
185     var params = '<param name="movie" value="'+ swfUrl +'" />';
186     for(var key in opt_params) {
187       if (key == 'width' || key == 'height') {
188         html += ' ' + key + '="' + opt_params[key] + '"';
189       } else {
190         params += '<param name="'+ key +'" value="'+ opt_params[key] +'" />';
191       }
192     }
193     html += '>' + params + "</object>";
194   }
195   container.innerHTML = html;
196   return true;
197 };

200 /**
201  * Returns the env object for container
202  * @requires gadgets.views
203  * @see //flash/actionscript/com/google/ccc/chinasocial/api/API.as
204  * @member flashAPI
205  * @return {Object} a map of all environment values.
206  */
207 flashAPI.getEnvironment = function() {
208   var envObj_ = {};
209   if (opensocial.getEnvironment) {
210     var env = opensocial.getEnvironment();
211     envObj_.domain = env.getDomain();
212     
213     // TODO Add support field check
214     
215   } else {
216     // TODO Here is work around!
217     envObj_.domain = flashAPI.extractDomain(document.referrer);
218   }
219   
220   // Default view type
221   envObj_.view = 'profile';
222     
223   if (gadgets) {
224     if (gadgets.views) {
225       envObj_.view = gadgets.views.getCurrentView().getName();
226     }
227   }
228   EXPORT('env', envObj_);  // for debugging
229   return envObj_;
230 };

266 /**
267  * Sends out the request for basic data set.
268  * NOTE:
269  * Here we pass JSON string instead of an object to Flash via the callback.
270  * See comments of 'handleFetchBasicData' method in the API.as file.
271  *
272  * @see //flash/actionscript/com/google/ccc/chinasocial/api/API.as
273  * @see //flash/actionscript/com/google/ccc/chinasocial/api/BasicData.as
274  * @member flashAPI
275  * 
276  * @param {string} reqID The request id.
277  * @param {Array.<string>} keys The app keys.
278  * @param {Object.<string, Object>} params 
279  *        A map of PeopleRequestField's names and values.
280  */
281 flashAPI.fetchBasicData = function(reqID, keys, params) {
282 
283   EXPORT('params_', params); // for debug
284 
285   params = flashAPI.wrapParams(
286       params, opensocial.DataRequest.PeopleRequestFields);
287 
288   // The params object need to be cloned, because req.add method will modify it!
289   var req = opensocial.newDataRequest();
290   req.add(req.newFetchPersonRequest('OWNER', 
291       flashAPI.cloneObject(params)), 'o');
292   req.add(req.newFetchPersonRequest('VIEWER', 
293       flashAPI.cloneObject(params)), 'v');
294   req.add(req.newFetchPeopleRequest('OWNER_FRIENDS', 
295       flashAPI.cloneObject(params)), 'of');
296   req.add(req.newFetchPeopleRequest('VIEWER_FRIENDS', 
297       flashAPI.cloneObject(params)), 'vf');
298   
299   /* TODO Not support by our container yet
300   req.add(req.newFetchPersonAppDataRequest('OWNER', keys), 'od');
301   req.add(req.newFetchPersonAppDataRequest('VIEWER', keys), 'vd');
302   */
303   
304   req.send(function(dataResponse) {
305     try {
306       EXPORT('raw', dataResponse); // for debug
307 
308       var basicData_ = new flashAPI.BasicData(dataResponse);
309 
310       EXPORT('basic_', basicData_); // for debug
311       
312       flashAPI.swfObj.handleFetchBasicData(reqID,
313           gadgets.json.stringify(basicData_));
314     } catch (e) {
315       flashAPI.swfObj.handleError(reqID, e);  
316     }
317   });
318 };

538 /**
539  * The collection class, wrapped from <code>opensocial.Collection</code>.
540  * @member flashAPI
541  * @class 
542  * @constructor
543  * @param {opensocial.Collection} collection The original data object.
544  * @param {Function} flashAPIType 
545  *        The constuctor of the items in this collection.
546  */
547 flashAPI.Collection = function(collection, flashAPIType) {
548   /**
549    * All the objects in this collection.
550    * @member flashAPI.Collection
551    * @type {Array}
552    */
553   this.array = [];
554 
555   /** 
556    * The total size of the larger result set that this collection belongs to.
557    * @member flashAPI.Collection
558    * @type {number}
559    */
560   this.totalSize = 0;
561   
562   /**
563    * The size of this collection, which is equal to or less than the total size 
564    * of the result.
565    * @member flashAPI.Collection
566    * @type {number}
567    */
568   this.size = 0;
569   
570   /**
571    * The offset of this collection within a larger result set.
572    * @member flashAPI.Collection
573    * @type {number}
574    */
575   this.offset = 0;
576 
577   // Initialization
578   if (!collection) return;
579   var array = null;
580   if (collection.asArray) { // due with real Collection
581     array = collection.asArray();
582     this.offset = collection.getOffset();
583     this.totalSize =  collection.getTotalSize();
584     this.size = collection.size();
585   } else if (flashAPI.isArray(collection)){ // due with array
586     array = collection;
587     this.offset = 0;
588     this.totalSize = collection.length;
589     this.size = this.totalSize;
590   }
591   if (array != null) {
592     for (var index in array) {
593       this.array.push(new flashAPIType(array[index]));
594     }
595   }
596 };

724 //--------------------- Data response accessors --------------------------------
725 /**
726  * Gets data object from resoponse.
727  * @member flashAPI
728  * @private
729  * @param {opensocial.DataResponse} dataResponse The response object.
730  * @param {string} key The data key in the response object.
731  * @throws {Error} When the RPC meets error.
732  * @return {Object?} The data, can be any instance of the opensocial data type,
733  *                   or a collection data type.
734  */
735 flashAPI.getData = function(dataResponse, key) {
736   if (!dataResponse) return null;
737   if (dataResponse.hadError()) {
738     console.info('[ERROR] Data Response Error!');
739   }
740   if (!key) return null;
741   var responseItem = dataResponse.get(key);
742   if (!responseItem) return null;
743   if (responseItem.hadError()) {

836 //--------------------- Simple type wrappers ---------------------------------
837 /**
838  * Checks the responded opensocial data structure's type.
839  * @member flashAPI
840  * @private
841  * @param {Object?} dataObj The original opensocial data object.
842  * @return {Function?} the types' constructor.
843  */
844 flashAPI.checkType = function(dataObj) {
845   var type = null;
846   // TODO: MediaItem and BodyType are not supported yet.
847   if (opensocial.Address && dataObj instanceof opensocial.Address) {
848     type = opensocial.Address;
849   } else if (opensocial.Email && dataObj instanceof opensocial.Email) {
850     type = opensocial.Email;
851   } else if (opensocial.Name && dataObj instanceof opensocial.Name) {
852     type = opensocial.Name;
853   } else if (opensocial.Organization && 
854              dataObj instanceof opensocial.Organization) {
855     type = opensocial.Organization;
856   } else if (opensocial.Phone && dataObj instanceof opensocial.Phone) {
857     type = opensocial.Phone;
858   } else if (opensocial.Url && dataObj instanceof opensocial.Url) {
859     type = opensocial.Url;
860   } else if (opensocial.Person && dataObj instanceof opensocial.Person) {
861     type = opensocial.Person;
862   } else if (opensocial.Activity && dataObj instanceof opensocial.Activity) {
863     type = opensocial.Activity;
864   } else if (opensocial.Message && dataObj instanceof opensocial.Message) {
865     type = opensocial.Message;
866   } else if (opensocial.Enum && dataObj instanceof opensocial.Enum) {
867     type = opensocial.Enum;
868   }
869   return type;
870 };

873 /**
874  * Wraps the data objects fields to JSON-style objects without any 
875  * getter/setter functions. Because in actionscript's ExternalInterface cannot 
876  * use the getFields to get the data correctly.
877  * @member flashAPI
878  * @private
879  * @param {Object?} dataObj The opensocial data, 
880  *        e.g. <code>opensocial.Email</code>, <code>opensocial.Person</code>...
881  * @param {Object?} opt_this The converted obj for flash.
882  * @return {Object?} the converted obj for flash.
883  */
884 flashAPI.wrapFields = function(dataObj, opt_this) {
885   if (dataObj == null) return null;
886   var type = flashAPI.checkType(dataObj);
887 
888   if (type == null) {
889     // treat it as string
890     return dataObj;
891   } else if (type == opensocial.Enum) {
892     // return enum name
893     return dataObj.getDisplayValue();
894   }
895 
896   // Null parameter for preventing extra recursion.
897   opt_this = opt_this || new flashAPI.SimpleType(null);    
898 
899   opt_this.fields = opt_this.fields || {};
900 
901   // Grabs the fields data from each structures recursively.
902   for (var k in type.Field) {
903     var key = type.Field[k];
904     var value = dataObj.getField(key);
905     if (value != null) {
906       if (flashAPI.isArray(value)) {
907         opt_this.fields[k] = [];
908         flashAPI.forEach(value, function(item) {
909           opt_this.fields[k].push(flashAPI.wrapFields(item));
910         });
911       } else {
912         opt_this.fields[k] = flashAPI.wrapFields(value);
913       }
914     }
915   }
916 
917   return opt_this || {};
918 };

921 /**
922  * Wraps JOSN-style (raw) params object (from actionscript) keys to opensocial 
923  * field types.
924  * TODO: Maybe more field types in the futures. Not neatly imeplemtation here.
925  * @member flashAPI
926  * @private
927  * @param {Object.<string, Object>?} rawParams 
928  *        The params object from actionscript.
929  * @param {Function?} type 
930  *        The field type class constructor.
931  *        e.g. <code>opensocial.DataRequest.PeopleRequestFields</code>.
932  * @return {Object.<Object, Object>?} New params with the correct keys.
933  */
934 flashAPI.wrapParams = function(rawParams, type) {
935   if (!type || !rawParams) return null;
936   var params = {};
937   for (var strKey in rawParams) {
938     switch (strKey) {
939       case 'PROFILE_DETAILS':
940         var fieldNames = rawParams[strKey];
941         if (fieldNames != null) {
942           var fieldArray = [];
943           // Convert string array to request field objects array
944           for (var i = 0; i < fieldNames.length; ++i) {
945             fieldArray.push(opensocial.Person.Field[fieldNames[i]]);
946           }
947           params[type[strKey]] = fieldArray;
948         }
949         break;
950       case 'FILTER':
951         params[type[strKey]] = 
952             opensocial.DataRequest.FilterType[rawParams[strKey]];
953         break;
954       case 'SORT_ORDER':
955         params[type[strKey]] = 
956             opensocial.DataRequest.SortOrder[rawParams[strKey]];
957         break;
958       default:
959         params[type[strKey]] = rawParams[strKey];
960         break;
961     }
962   }
963   return params;
964 }

970  * @member flashAPI
971  * @private
972  * @param {Object} obj object to be check.
973  * @return {boolean} True if the object is an array.

975 flashAPI.isArray = function(obj) {
976   if (!obj) return false;
977   if (Array) {
978     return obj instanceof Array;
979   } else {
980     if ((typeof obj == 'object') && 
981         (typeof obj.length == 'number') && 
982         (typeof obj.splice != 'undefined')) {
983       return true;
984     } else {
985       return false;
986     }
987   }

990 /**
991  * Iterates an array, calls the function for each item.
992  * @member flashAPI
993  * @private
994  * @param {Array} array The array to be iterated.
995  * @param {Function} f The function to be applied.
996  * @param {Object} opt_thisObj The this object for the function.

998 flashAPI.forEach = function(array, f, opt_thisObj) {
999   var l = array.length;
1000   for (var i = 0; i < l; i++) {
1001     if (i in array) {
1002       f.call(opt_thisObj, array[i], i, array);
1003     }
1004   }

1024 /**
1025  * Deeply clones an object.
1026  * @member flashAPI
1027  * @private
1028  * @param {Object?} obj The original object.
1029  * @return {Object?} The cloned object.
1030  */
1031 flashAPI.cloneObject = function(obj) {
1032   if(obj == null) return null;
1033   switch (typeof obj){
1034     case 'function':
1035       return obj;
1036     case 'object': 
1037       var newObj = new obj.constructor();
1038       for(var key in obj) newObj[key] = flashAPI.cloneObject(obj[key]);
1039       return newObj;
1040     default:
1041       return obj.valueOf();
1042   }
1043 };

1061 /**
1062  * For debugging, return the object's keys and values.
1063  * @member flashAPI
1064  * @private
1065  * @param {Object?} obj Any object to be looked into.
1066  * @param {string?} opt_prefix The prefix string of each output line.
1067  * @param {string?} opt_rendered The rendered string passing between recursions.
1068  * @return {string?} A tree-style string to represent the object.
1069  */
1070 flashAPI.debugInspect = function(obj, opt_prefix, opt_rendered) {
1071   if (!opt_rendered) opt_rendered = '\n';
1072   if (!opt_prefix) opt_prefix = '';
1073   if (obj == null) return opt_rendered || "";
1074   for (var k in obj) {
1075     opt_rendered += 
1076         (opt_prefix + k + '[' + (typeof obj[k]) + '] : ' + obj[k] + '\n');
1077     if ((typeof obj[k]) == 'object') {
1078       opt_rendered = 
1079           flashAPI.debugInspect(obj[k], opt_prefix + '\t', opt_rendered);
1080     }
1081   }
1082   return opt_rendered || "";
1083 };
