 90   /**
 91    * For debugging, return the object's keys and values.
 92    * @param {Object?} obj Any object to be looked into.
 93    * @param {string?} opt_prefix The prefix string of each output line.
 94    * @param {string?} opt_rendered The rendered string passing between recursions.
 95    * @return {string?} A tree-style string to represent the object.
 96    * @private
 97    */
 98   var INSPECT = function(obj, opt_prefix, opt_rendered) {
 99     if (!opt_rendered) opt_rendered = '\n';
100     if (!opt_prefix) opt_prefix = '';
101     if (obj == null) return opt_rendered || "";
102     for (var k in obj) {
103       opt_rendered +=
104           (opt_prefix + k + '[' + (typeof obj[k]) + '] : ' + obj[k] + '\n');
105       if ((typeof obj[k]) == 'object') {
106         opt_rendered = INSPECT(obj[k], opt_prefix + '\t', opt_rendered);
107       }
108     }
109     return opt_rendered || "";
110   };

238      * Checks if the object is an array.
239      * @param {Object} obj object to be check.
240      * @return {boolean} True if the object is an array.
241      * @private

243     me.isArray = function(obj) {
244       if (!obj) return false;
245       if (Array) {
246         return obj instanceof Array;
247       } else {
248         if ((typeof obj == 'object') &&
249             (typeof obj.length == 'number') &&
250             (typeof obj.splice != 'undefined')) {
251           return true;
252         } else {
253           return false;
254         }
255       }

258     /**
259      * Iterates an array, calls the function for each item.
260      * @param {Array} array The array to be iterated.
261      * @param {Function} f The function to be applied.
262      * @param {Object} opt_thisObj The this object for the function.
263      * @private
264      */
265     me.forEach = function(array, f, opt_thisObj) {
266       if (typeof array.length != 'number') {
267         return;
268       }
269       var l = array.length;
270       for (var i = 0; i < l; i++) {
271         if (i in array) {
272           if (false == f.call(opt_thisObj, array[i], i, array)) {
273             break;
274           };
275         }
276       }
277     };

280     /**
281      * Deeply clones an non-native object.
282      * @param {Object?} obj The original object.
283      * @return {Object?} The cloned object.
284      * @private
285      */
286     me.clone = function(obj) {
287       if(obj == null) return null;
288       switch (typeof obj){
289         case 'function':
290           return obj;
291         case 'object':
292           var newObj = new obj.constructor();
293           for(var key in obj) newObj[key] = me.clone(obj[key]);
294           return newObj;
295         default:
296           return obj.valueOf();
297       }

300     //======================== Opensocial Data Helper Functions (private) ==========================
301     /**
302      * Gets data object from resoponse.
303      * @param {opensocial.DataResponse} dataResponse The response object.
304      * @param {string} key The data key in the response object.
305      * @throws {Error} When the RPC meets error.
306      * @return {Object?} The data, can be any instance of the opensocial.* data type.
307      * @private
308      */
309     me.getData = function(dataResponse, key) {
310       // for debug
311       EXPORT('DATA_RESPONSE', {'key': key, 'data': dataResponse});
312 
313       if (!dataResponse) return null;
314       if (dataResponse.hadError()) {
315         var e = new Error(dataResponse.getErrorMessage());
316         e['name'] = "OpensocialError";
317         throw e;
318       }
319       if (!key) return null;
320 
321       return me.getDataItem(dataResponse.get(key));
322     };

324     /**
325      * Gets data object from response item.
326      * @param {opensocial.ResponseItem} responseItem The response item object.
327      * @throws {Error} When the items have errors.
328      * @return {Object?} The data, can be any instance of the opensocial.* data type.
329      * @private
330      */
331     me.getDataItem = function(responseItem) {
332       if (!responseItem) return null;
333       if (responseItem.hadError()) {
334         var e = new Error(responseItem.getErrorMessage());
335         e['name'] = "OpensocialError";
336         e['code'] = responseItem.getErrorCode();
337         throw e;
338       }
339       return responseItem.getData();
340     };

342     me.dataTypes = ['Address', 'BodyType', 'Email', 'Name', 'Organization', 'Phone', 'Url',
343         'Enum', 'Person', 'Activity', 'IdSpec', 'MediaItem', 'Message', 'NavigationParameters',
344         'Collection'];
345     /**
346      * Gets the constructor function for opensocial data object.
347      * @param {Object} dataObj The object to be checked.
348      * @return {Function?} The type of the object.
349      * @private
350      */
351     me.checkType = function(dataObj) {
352       if (!dataObj) return null;
353       var type = null;
354       me.forEach(me.dataTypes, function(typeItem) {
355         if (typeItem in opensocial && dataObj instanceof opensocial[typeItem]) {
356           type = opensocial[typeItem];
357           return false;
358         }
359       });
360       return type;

363     /**
364      * Wraps the opensocial rich object (with functions) to a JSON-like flat serializable object.
365      * So the object can pass through the <code>ExternalInterface</code> to the swf.
366      * @param {Object} dataObj The object to be wrapped.
367      * @param {Object?} opt_this The flat object as a container to hold the wrapped fields.
368      * @return {Object?} A wrapped flat object, ready to pass to swf.
369      * @private
370      */
371     me.wrapObject = function(dataObj, opt_this) {
372       if (dataObj == null) return null;
373 
374       var type = me.checkType(dataObj);
375 
376       if (type == null) {
377         // treat it as string
378         return dataObj;
379       }
380 
381       // fix fields
382       opt_this = opt_this || {};
383 
384       if ('Field' in type) {
385         opt_this['fields'] = opt_this['fields'] || {};
386 
387         // Grabs the fields data from each structures recursively.
388         for (var k in type.Field) {
389           var key = type.Field[k];
390           var value = dataObj.getField(key);
391           if (value != null) {
392             if (me.isArray(value)) {
393               opt_this['fields'][key] = [];
394               me.forEach(value, function(item) {
395                 opt_this['fields'][key].push(me.wrapObject(item));
396               });
397             } else {
398               opt_this['fields'][key] = me.wrapObject(value);
399             }
400           }
401         }
402       }
403 
404       // fix properties.
405       if (type == opensocial.Person) {
406         opt_this['displayName'] = dataObj.getDisplayName();
407         opt_this['isOwner'] = dataObj.isOwner();
408         opt_this['isViewer'] = dataObj.isViewer();
409         opt_this['id'] = dataObj.getId();
410 
411       } else if (type == opensocial.Activity) {
412         opt_this['id'] = dataObj.getId();
413 
414       } else if (type == opensocial.Enum) {
415         opt_this['key'] = dataObj.getKey();
416         opt_this['displayValue'] = dataObj.getDisplayValue();
417 
418       } else if (type == opensocial.Collection) {
419         opt_this['size'] = dataObj.size();
420         opt_this['offset'] = dataObj.getOffset();
421         opt_this['totalSize'] = dataObj.getTotalSize();
422 
423         var array = dataObj.asArray();
424         opt_this['array'] = [];
425         me.forEach(array, function(item) {
426           opt_this['array'].push(me.wrapObject(item));
427         });
428 
429       }
430       return opt_this || {};
431     };

433     /**
434      * Unwraps the flattened object to a rich opensocial data object. The flattened object is come
435      * from the swf via <code>ExternalInterface</code>.
436      * @param {Object} obj The object to be unwrapped. Only those 5 types with <code>setField</code>
437      *                 method and <code>opensocial.newXXXX</code> creator can be unwrapped.
438      * @param {Function} type The object's target type to be unwrapped.
439      * @throws {Error} When the type is not correct.
440      * @return {Object} Unwrapped rich opensocial data object, ready to use in javascript.
441      * @private
442      */
443     me.unwrapObject = function(obj, type) {
444       if (obj && 'fields' in obj) {
445         var fields = obj['fields'];
446 
447         if (type == opensocial.IdSpec) {
448           return opensocial.newIdSpec(fields);
449 
450         } else if (type == opensocial.NavigationParameters) {
451           return opensocial.newNavigationParameters(fields);
452 
453         } else if (type == opensocial.MediaItem) {
454           return opensocial.newMediaItem(fields[opensocial.MediaItem.Field.MIME_TYPE],
455                                          fields[opensocial.MediaItem.Field.URL],
456                                          fields);
457 
458         } else if (type == opensocial.Message) {
459           return opensocial.newMessage(fields[opensocial.Message.Field.BODY],
460                                        fields);
461 
462         } else if (type == opensocial.Activity) {
463           var flatItems = fields[opensocial.Activity.Field.MEDIA_ITEMS];
464           if (me.isArray(flatItems)) {
465             var richItems = [];
466             me.forEach(flatItems, function(item) {
467               richItems.push(me.unwrapObject(item, opensocial.MediaItem));
468             });
469             fields[opensocial.Activity.Field.MEDIA_ITEMS] = richItems;
470           }
471           return opensocial.newActivity(fields);
472         }
473 
474       }
475       // If reaches here, codes are wrong.
476       var e = new Error('The object does not has fields values or type mismatched.');
477       e['name'] = "OpensocialError";
478       throw e;
479     };

515     /**
516      * Pushes the running rpc instance to the store and returns an unique callID. It is used by

548       /**
549        * Indicates if the opensocial library is loaded and initialize the flash element.
550        * <p>
551        * The objectId of the swf is defined in gagdet spec. If the swf is embeded by calling
552        * <code>opensocial.flash.embedFlash</code> function, the detault objectId will be 'flashObj'.
553        * </p><p>
554        * This function is called by the swf's <code>ExternalInterface</code> continuingly until it
555        * returns true.
556        * </p>
557        *
558        * @param {string?} objectId The object Id passed from the swf.
559        * @return {boolean} true if all javascripts are loaded.
560        *
561        * @member opensocial.flash
562        */
563       jsReady : function(objectId) {
564         try {
565           if (!me.swfObj) {
566             me.swfObj = me.thisMovie(objectId);
567           }
568           return !!me.swfObj;
569         } catch (e) {
570           INFO(e);
571           return false;
572         }

575       /**
576        * Because <code>gadgets.flash.embedFlash</code> cannot auto activate it in IE6, so here we
577        * need a special one.
578        * <p>
579        * NOTE: to be deprecated.
580        * </p>
581        *
582        * @requires gadgets.flash
583        *
584        * @param {string} swfUrl  The swf file's location.
585        * @param {string} swfContainer  The HTML element id.
586        * @param {string?} opt_playerVersion Required flash player version.
587        * @param {Object.<string, string>?} opt_params Flash object's parameters.
588        * @param {string?} opt_upgradeMsgHTML The message if the flashplayer version is too old.
589        * @return {boolean} True if the flash object is successfully embeded.
590        *
591        * @member opensocial.flash
592        */
593       embedFlash : function(swfUrl, swfContainer,
594                             opt_playerVersion, opt_params, opt_upgradeMsgHTML) {
595 
596         opt_params = opt_params || {};
597 
598         // Check container element
599         var container = me.thisDoc().getElementById(swfContainer);
600         if (!container) return false;
601 
602         // Check version
603         var version = opt_playerVersion.match(/(\d+)(?=[\.,]?)/g);
604         var expectMajorVer = (version != null ? Number(version[0]) : 9);
605 
606         EXPORT("PLAYER_CURRENT_VER", gadgets.flash.getMajorVersion());
607         EXPORT("PLAYER_EXPECED_VER", expectMajorVer);
608 
609         if (gadgets.flash.getMajorVersion() < expectMajorVer) {
610           opt_upgradeMsgHTML = opt_upgradeMsgHTML ||
611               '<span style="font-size:small;color:#777;"> ' +
612               'Please install or upgrade to ' +
613               '<a href="http://www.adobe.com/cn/products/flashplayer/" ' +
614               'style="color:#36C;" target="_blank">Flash Player 9.0</a>.</span>';
615           container.innerHTML = opt_upgradeMsgHTML;
616           return false;
617         };
618 
619         /*
620         opt_params['id'] = me.DEFAULT_FLASH_ID;
621         opt_params['name'] = me.DEFAULT_FLASH_ID;
622         return gadgets.flash.embedFlash(swfUrl, container, version, opt_params);
623         */
624         // Generate the flash object element
625         var html = "";
626         if (me.thisBrowser().plugins &&
627             me.thisBrowser().mimeTypes &&
628             me.thisBrowser().mimeTypes.length) {
629           // netscape plugin architecture
630           html = '<embed src="'+ swfUrl +
631                  '" id="' + me.DEFAULT_FLASH_ID +
632                  '" name="' + me.DEFAULT_FLASH_ID +
633                  '" type="application/x-shockwave-flash"';
634           for(var key in opt_params){
635             html += key +'="'+ opt_params[key] +'" ';
636           }
637           html += '/>';
638         } else {
639           // PC IE
640           html = '<object id="' + me.DEFAULT_FLASH_ID +
641                  '" classid="clsid:D27CDB6E-AE6D-11cf-96B8-444553540000"';
642           var params = '<param name="movie" value="'+ swfUrl +'" />';
643           for(var key in opt_params) {
644             if (key == 'width' || key == 'height') {
645               html += ' ' + key + '="' + opt_params[key] + '"';
646             } else {
647               params += '<param name="'+ key +'" value="'+ opt_params[key] +'" />';
648             }
649           }
650           html += '>' + params + "</object>";
651         }
652         container.innerHTML = html;
653 
654         return true;
655 
656       },

1013        * @param {Object.<gadgets.io.RequestParameters, Object>?} opt_params
1014        *            Parameters for the request
1015        * @member opensocial.flash
1016        */
1017       makeRequest : function(reqID, url, opt_params) {
1018         opt_params = opt_params || {}
1019         var postData = opt_params[gadgets.io.RequestParameters.POST_DATA];
1020         if (postData) {
1021           opt_params[gadgets.io.RequestParameters.POST_DATA] = gadgets.io.encodeValues(postData);
1022         }
1023         gadgets.io.makeRequest(url, function(obj) {
1024           try {
1025             if (obj['error'] || obj['rc'] != 200) {
1026               var err = new Error(obj['text']);
1027               err['name'] = 'OpensocialError';
1028               throw err;
1029             }
1030             me.handleAsync(reqID, obj.data);
1031           } catch (e) {
1032             me.handleError(reqID, e);
1033           }
1034         }, opt_params);

1037       /**
1038        * Sends the rpc call request to a service on another app or container on the page.
1039        *
1040        * @param {string} reqID The request id.
1041        * @param {string?} targetId The target app id, null to call the parent.
1042        * @param {string} serviceName The service name registered on the target.
1043        * @param {Array.<Object>?} argsArray The array of args to be passed to the service.
1044        * @member opensocial.flash
1045        */
1046       callRpc : function(reqID, targetId, serviceName, argsArray) {
1047         var reqArgs = [targetId, serviceName, function() {
1048           if (arguments.length > 0) {
1049             me.handleAsync(reqID, arguments[0]);
1050           } else {
1051             me.handleAsync(reqID);
1052           }
1053         }];
1054         if (argsArray) {
1055           me.forEach(argsArray, function(arg) {
1056             reqArgs.push(arg);
1057           });
1058         }
1059         gadgets.rpc.call.apply(null, reqArgs);
1060       },

1075       /**
1076        * Sends the rpc register request to container to register a service on this app.
1077        *
1078        * @param {string} serviceName The service to be register.
1079        * @param {string} handlerName The handler function name on the flash object.
1080        * @member opensocial.flash
1081        */
1082       registerRpcService : function(serviceName, handlerName) {
1083         var handler = function() {
1084           try {
1085             var callID = me.pushRpc(this);
1086             var args = [callID];
1087             me.forEach(arguments, function(arg) {
1088               args.push(arg);
1089             });
1090             me.swfObj[handlerName].apply(me.swfObj, args);
1091             // no return here, call rpcReturn in flash to return the result.
1092           } catch(e) {
1093             me.handleError(null, e);
1094             return null;
1095           }
1096         };
1097         
1098         if (serviceName == null) {
1099           gadgets.rpc.registerDefault(handler);
1100         } else {
1101           gadgets.rpc.register(serviceName, handler);
1102         }
1103       },
